### 4g. Recall — ScoringWeights

```python
from hebbs import ScoringWeights

async def test_recall_weights():
    async with HebbsClient("localhost:6380") as h:
        # Recency-heavy weights
        recency_weights = ScoringWeights(
            w_relevance=0.1, w_recency=0.7, w_importance=0.1, w_reinforcement=0.1
        )
        result_recency = await h.recall(
            cue="Initech evaluation",
            scoring_weights=recency_weights,
            top_k=5,
        )

        # Relevance-heavy weights
        relevance_weights = ScoringWeights(
            w_relevance=0.8, w_recency=0.05, w_importance=0.1, w_reinforcement=0.05
        )
        result_relevance = await h.recall(
            cue="Initech evaluation",
            scoring_weights=relevance_weights,
            top_k=5,
        )

        print("=== Recency-heavy ===")
        for r in result_recency.results[:3]:
            print(f"  {r.score:.3f}  {r.memory.content[:50]}")

        print("=== Relevance-heavy ===")
        for r in result_relevance.results[:3]:
            print(f"  {r.score:.3f}  {r.memory.content[:50]}")

asyncio.run(test_recall_weights())
```

- [ ] Both queries succeed
- [ ] Ordering differs between recency-heavy and relevance-heavy (most recent memories ranked higher with recency weights)
- [ ] Scores reflect the weight configuration
- [ ] Passing `ScoringWeights` as a dataclass works

### 4j. Recall — Causal Strategy

```python
async def test_recall_causal():
    async with HebbsClient("localhost:6380") as h:
        # First, find a memory to use as seed
        sim = await h.recall(cue="Initech CTO", strategies=["similarity"], top_k=1, entity_id="initech")
        if sim.results:
            seed_id = sim.results[0].memory.id
            result = await h.recall(
                cue="Initech",
                strategies=[
                    RecallStrategyConfig(
                        strategy="causal",
                        seed_memory_id=seed_id,
                        max_depth=3,
                        edge_types=[EdgeType.FOLLOWED_BY, EdgeType.CAUSED_BY],
                    ),
                ],
            )
            print(f"causal results={len(result.results)}")
            for r in result.results:
                for d in r.strategy_details:
                    print(f"  depth={d.depth}  {r.memory.content[:50]}")
        else:
            print("No seed memory found — skip causal test")

asyncio.run(test_recall_causal())
```

- [ ] `depth` in strategy details is populated

import asyncio

async def test_subscribe():
    async with HebbsClient("localhost:6380") as h:
        sub = await h.subscribe(entity_id="initech", confidence_threshold=0.3)
        print(f"subscription opened: id={sub.subscription_id}")

        # Feed some context
        await sub.feed("Tell me about Initech's evaluation process")
        print("fed context, waiting for pushes...")

        # Collect pushes for 3 seconds
        pushes = []
        try:
            async for push in sub.listen(timeout=3.0):
                pushes.append(push)
                print(f"  push: confidence={push.confidence:.2f}  {push.memory.content[:50]}")
        except asyncio.TimeoutError:
            pass

        print(f"total pushes: {len(pushes)}")

        await sub.close()
        print("subscription closed")

asyncio.run(test_subscribe())

| `HEBBS_REFLECT_VALIDATION_MODEL` | Same options as proposal |
| `HEBBS_EMBEDDING_PROVIDER` | `onnx` (default) or `mock` |
| `HEBBS_STORAGE_DATA_DIR` | Data directory (default: `./hebbs-data`) |
